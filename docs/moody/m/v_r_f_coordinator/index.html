                'gas': _gaswei,
                'gasPrice': _pricewei
            })
            _t['nonce'] = self._web3_eth.getTransactionCount(self._operate)

            print(f"{Bolors.FAIL}Error {er} {Bolors.RESET}: withdraw")

        except ValueError as err:
